cmake_minimum_required(VERSION 3.8 FATAL_ERROR)
project (kmeans LANGUAGES CUDA CXX)
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -pedantic -Wextra")

# Included headers #
include_directories(include)

# Util lib
add_library(util ${PROJECT_SOURCE_DIR}/src/util.cpp)

# Cuda lib
add_library(cuda_kmeans STATIC ${PROJECT_SOURCE_DIR}/src/kmeans.cu)
set_target_properties(cuda_kmeans PROPERTIES CUDA_SEPARATE_COMPILATION ON CUDA_STANDARD 11)

# Add sources #
set(SOURCES ${PROJECT_SOURCE_DIR}/src/main.cpp)

# Program binaries and linker
add_executable(kmeans ${SOURCES})
set_property(TARGET kmeans PROPERTY CUDA_SEPARATE_COMPILATION ON)
target_link_libraries(kmeans PUBLIC util cuda_kmeans libdocopt graph ${CMAKE_THREAD_LIBS_INIT})
install(TARGETS kmeans RUNTIME DESTINATION bin)
